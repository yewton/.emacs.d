(defun toncs-dired-add-entry (oldfun &rest args)
  "`dirvish-subtree' で展開したディレクトリは `dired-subdir-alist' に
登録されないため、そこへドラッグ&ドロップ等でファイルを追加すると
`dired-add-entry' が \"This cannot happen\" エラーを送出する
（ファイル自体は正しい場所に配置済み）。その場合は差分更新を諦めて
バッファ全体を revert する。無関係なエラーまで握り潰さないよう、
サブツリー展開中かつこのエラーメッセージのときだけフォールバックする。"
  (condition-case err
      (apply oldfun args)
    (error
     (if (and (bound-and-true-p dirvish-subtree--overlays)
              (equal (error-message-string err) "This cannot happen"))
         (revert-buffer)
       (signal (car err) (cdr err))))))

(defun toncs-config-dired-configure ()
  (setq dired-dwim-target t)
  (setq dired-recursive-deletes 'always)
  (setq delete-by-moving-to-trash t)
  (defvar dired-bind-jump nil) ; 何かの拍子に dired-x ロードされても、 C-x C-j は SKK の為に取っておきたい
  (let ((gls (executable-find "gls")))
    (when gls
      (setq insert-directory-program gls)))
  (setq dired-listing-switches "-l --almost-all --human-readable --time-style=long-iso --group-directories-first --no-group")

  (advice-add 'dired-add-entry :around #'toncs-dired-add-entry)

  (dirvish-override-dired-mode 1)
  (dirvish-peek-mode 1)
  (setq dirvish-peek-key (list :debounce 0.5 'any))
  (setq dirvish-attributes '(vc-state subtree-state all-the-icons collapse git-msg file-size))
  (setq dirvish-mode-line-format '(:left (sort file-time " " file-size symlink) :right (omit yank index)))
  (setq mouse-1-click-follows-link nil)
  (general-def dirvish-mode-map
    "a" #'dirvish-quick-access
    "f" #'dirvish-file-info-menu
    "y" #'dirvish-yank-menu
    "N" #'dirvish-narrow
    "h" #'dirvish-history-jump ; remapped `describe-mode'
    "s" #'dirvish-quicksort    ; remapped `dired-sort-toggle-or-edit'
    "v" #'dirvish-vc-menu      ; remapped `dired-view-file'
    "r" #'wdired-change-to-wdired-mode
    "TAB" #'dirvish-subtree-toggle
    "M-f" #'dirvish-history-go-forward
    "M-b" #'dirvish-history-go-backward
    "M-l" #'dirvish-ls-switches-menu
    "M-m" #'dirvish-mark-menu
    "M-t" #'dirvish-layout-toggle
    "M-e" #'dirvish-emerge-menu
    "<mouse-1>" #'dirvish-subtree-toggle-or-open
    "<mouse-2>" #'dired-mouse-find-file-other-window
    "<mouse-3>" #'dired-mouse-find-file)
  (general-def
    "<f12>" #'dirvish-side)

  (dirvish-define-preview eza (file)
    "Use `exa' to generate directory preview."
    :require ("eza") ; tell Dirvish to check if we have the executable
    (when (file-directory-p file) ; we only interest in directories here
      `(shell . ("eza" "-al" "--color=always" "--icons"
                 "--group-directories-first" ,file))))

  (add-to-list 'dirvish-preview-dispatchers 'eza)

  (add-hook 'dired-mode-hook #'org-download-enable))
