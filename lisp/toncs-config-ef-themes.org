#+begin_src emacs-lisp
(require 'modus-themes)
(require 'ef-themes)
(require 'general)

(defun toncs-config-ef-themes-configure ()
  (setq modus-themes-headings
      '((1 variable-pitch 1.3)
        (2 variable-pitch 1.2)
        (agenda-date 1.3)
        (agenda-structure variable-pitch light 1.3)
        (t variable-pitch 1.1)))
  (setq modus-themes-common-palette-overrides
        '((bg-region bg-cyan-intense)
          (fg-region fg-intense)))
  (setq modus-themes-mixed-fonts t
        modus-themes-variable-pitch-ui t)
  (mapc #'disable-theme custom-enabled-themes)
  (setq modus-themes-to-toggle '(ef-duo-dark ef-duo-light))
  (modus-themes-include-derivatives-mode +1)
  (modus-themes-load-theme 'ef-duo-dark)
  ;; こうしないと、部分的にテーマが反映されなくなる
  (add-hook 'emacs-startup-hook (lambda () (modus-themes-load-theme 'ef-duo-dark)))
  (general-def "<f9>" #'modus-themes-toggle))
#+end_src
